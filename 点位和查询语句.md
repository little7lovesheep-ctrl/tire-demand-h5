# 点位统计

## 表结构

| 字段 | 类型 | 可空 | 默认值 | 说明 |
| --- | --- | --- | --- | --- |
| `id` | `int(11)` | 否 | 无 | 主键、自增 ID |
| `phone` | `varchar(32)` | 是 | 空字符串 | 分享者账号 |
| `page` | `varchar(50)` | 是 | `NULL` | 页面 |
| `act` | `varchar(200)` | 是 | `NULL` | 点位名称 |
| `time` | `timestamp` | 否 | `CURRENT_TIMESTAMP` | 上报时间 |
| `ip` | `varchar(64)` | 否 | 空字符串 | 访问 IP |
| `source` | `varchar(64)` | 否 | 空字符串 | 来源 |
| `platform` | `varchar(64)` | 是 | `NULL` | 数据来源：`app`、`wechat`、`web` |
| `uid` | `int(11)` | 是 | `NULL` | 用户 ID，UV 唯一标识 |
| `openid` | `varchar(255)` | 是 | `NULL` | 微信用户标识 |
| `uuid` | `varchar(64)` | 是 | 空字符串 | 匿名用户标识 |
| `descr` | `varchar(50)` | 是 | 空字符串 | 备注 |
| `activity_id` | `int(11) unsigned` | 否 | `0` | 活动 ID |
| `referral_code` | `varchar(16)` | 否 | 空字符串 | 推荐码 |
| `track_id` | `varchar(64)` | 否 | 空字符串 | 单次会话标识 |
| `ext_info` | `text` | 是 | `NULL` | 扩展数据 |

## 按日统计 PV/UV

每个日期只生成一行，先按主漏斗排列，其他点位放在后面。UV 以 `uid` 去重，`uid IS NULL` 不计入 UV。

```sql
SELECT
    DATE(`time`) AS `日期`,
    SUM(`act` = '进入活动页') AS `进入活动页_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '进入活动页' THEN `uid` END) AS `进入活动页_UV`,
    SUM(`act` = '页面停留10秒') AS `页面停留10秒_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '页面停留10秒' THEN `uid` END) AS `页面停留10秒_UV`,
    SUM(`act` = '用户主动滑动') AS `用户主动滑动_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '用户主动滑动' THEN `uid` END) AS `用户主动滑动_UV`,
    SUM(`act` = '点击页面返回') AS `点击页面返回_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '点击页面返回' THEN `uid` END) AS `点击页面返回_UV`,
    SUM(`act` = '点击立即选购') AS `点击立即选购_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '点击立即选购' THEN `uid` END) AS `点击立即选购_UV`,
    SUM(`act` = '浏览到确认轮胎和数量') AS `浏览到确认轮胎和数量_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '浏览到确认轮胎和数量' THEN `uid` END) AS `浏览到确认轮胎和数量_UV`,
    SUM(`act` = '点击减少轮胎数量') AS `点击减少轮胎数量_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '点击减少轮胎数量' THEN `uid` END) AS `点击减少轮胎数量_UV`,
    SUM(`act` = '点击增加轮胎数量') AS `点击增加轮胎数量_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '点击增加轮胎数量' THEN `uid` END) AS `点击增加轮胎数量_UV`,
    SUM(`act` = '选择安装门店') AS `选择安装门店_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '选择安装门店' THEN `uid` END) AS `选择安装门店_UV`,
    SUM(`act` = '选择预计到店时间') AS `选择预计到店时间_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '选择预计到店时间' THEN `uid` END) AS `选择预计到店时间_UV`,
    SUM(`act` = '浏览到支付汇总') AS `浏览到支付汇总_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '浏览到支付汇总' THEN `uid` END) AS `浏览到支付汇总_UV`,
    SUM(`act` = '点击确认并在线支付') AS `点击确认并在线支付_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '点击确认并在线支付' THEN `uid` END) AS `点击确认并在线支付_UV`,
    SUM(`act` = '打开支付确认弹窗') AS `打开支付确认弹窗_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '打开支付确认弹窗' THEN `uid` END) AS `打开支付确认弹窗_UV`,
    SUM(`act` = '点击确认支付') AS `点击确认支付_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '点击确认支付' THEN `uid` END) AS `点击确认支付_UV`,
    SUM(`act` = '表单提交') AS `表单提交_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '表单提交' THEN `uid` END) AS `表单提交_UV`,
    SUM(`act` = '创建支付订单') AS `创建支付订单_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '创建支付订单' THEN `uid` END) AS `创建支付订单_UV`,
    SUM(`act` = '创建支付订单成功') AS `创建支付订单成功_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '创建支付订单成功' THEN `uid` END) AS `创建支付订单成功_UV`,
    SUM(`act` = '跳转支付页面') AS `跳转支付页面_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '跳转支付页面' THEN `uid` END) AS `跳转支付页面_UV`,
    SUM(`act` = '展示二次确认支付结果弹窗') AS `展示二次确认支付结果弹窗_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '展示二次确认支付结果弹窗' THEN `uid` END) AS `展示二次确认支付结果弹窗_UV`,
    SUM(`act` = '自动确认支付结果') AS `自动确认支付结果_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '自动确认支付结果' THEN `uid` END) AS `自动确认支付结果_UV`,
    SUM(`act` = '点击确认支付结果') AS `点击确认支付结果_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '点击确认支付结果' THEN `uid` END) AS `点击确认支付结果_UV`,
    SUM(`act` = '支付成功') AS `支付成功_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '支付成功' THEN `uid` END) AS `支付成功_UV`,
    SUM(`act` = '表单提交成功') AS `表单提交成功_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '表单提交成功' THEN `uid` END) AS `表单提交成功_UV`,
    SUM(`act` = '展示支付成功订单详情') AS `展示支付成功订单详情_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '展示支付成功订单详情' THEN `uid` END) AS `展示支付成功订单详情_UV`,
    SUM(`act` = '选择未开放城市') AS `选择未开放城市_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '选择未开放城市' THEN `uid` END) AS `选择未开放城市_UV`,
    SUM(`act` = '手机号校验失败') AS `手机号校验失败_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '手机号校验失败' THEN `uid` END) AS `手机号校验失败_UV`,
    SUM(`act` = '提交未开放城市') AS `提交未开放城市_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '提交未开放城市' THEN `uid` END) AS `提交未开放城市_UV`,
    SUM(`act` = '关闭支付确认弹窗') AS `关闭支付确认弹窗_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '关闭支付确认弹窗' THEN `uid` END) AS `关闭支付确认弹窗_UV`,
    SUM(`act` = '点击返回修改') AS `点击返回修改_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '点击返回修改' THEN `uid` END) AS `点击返回修改_UV`,
    SUM(`act` = '表单提交失败') AS `表单提交失败_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '表单提交失败' THEN `uid` END) AS `表单提交失败_UV`,
    SUM(`act` = '支付成功后表单提交失败') AS `支付成功后表单提交失败_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '支付成功后表单提交失败' THEN `uid` END) AS `支付成功后表单提交失败_UV`,
    SUM(`act` = '创建支付订单失败') AS `创建支付订单失败_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '创建支付订单失败' THEN `uid` END) AS `创建支付订单失败_UV`,
    SUM(`act` = '发起支付失败') AS `发起支付失败_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '发起支付失败' THEN `uid` END) AS `发起支付失败_UV`,
    SUM(`act` = '支付未完成') AS `支付未完成_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '支付未完成' THEN `uid` END) AS `支付未完成_UV`,
    SUM(`act` = '支付结果查询失败') AS `支付结果查询失败_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '支付结果查询失败' THEN `uid` END) AS `支付结果查询失败_UV`,
    SUM(`act` = '关闭支付结果确认弹窗') AS `关闭支付结果确认弹窗_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '关闭支付结果确认弹窗' THEN `uid` END) AS `关闭支付结果确认弹窗_UV`,
    SUM(`act` = '关闭支付成功弹窗') AS `关闭支付成功弹窗_PV`,
    COUNT(DISTINCT CASE WHEN `act` = '关闭支付成功弹窗' THEN `uid` END) AS `关闭支付成功弹窗_UV`
FROM hcb.`hyb_page_log_ai`
WHERE `activity_id` = 62620
  AND `time` > '2026-06-28 23:59:59'
GROUP BY DATE(`time`)
ORDER BY `日期` DESC;
```
